INCLUDE_DIRECTORIES(
${CMAKE_SOURCE_DIR}
${CMAKE_SOURCE_DIR}/scribus
)

SET(BARCODEGENERATOR_PLUGIN_UI_SRC
barcodegeneratorbase.ui
)

SET(BARCODEGENERATOR_PLUGIN_MOC_CLASSES
barcode.h
barcodegenerator.h
)

SET(BARCODEGENERATOR_PLUGIN_SOURCES
barcode.cpp
barcodegenerator.cpp
)

SET(SCRIBUS_BARCODEGENERATOR_PLUGIN "barcodegenerator")

IF(QT_WRAP_UI)
  QT_WRAP_UI(${SCRIBUS_BARCODEGENERATOR_PLUGIN} BARCODEGENERATOR_PLUGIN_UI_H BARCODEGENERATOR_PLUGIN_UI_CPP ${BARCODEGENERATOR_PLUGIN_UI_SRC} )
ENDIF(QT_WRAP_UI)

IF(QT_WRAP_CPP)
  SCRIBUS_QT_WRAP_CPP(${BARCODEGENERATOR_PLUGIN_MOC_SOURCES} ${BARCODEGENERATOR_PLUGIN_MOC_CLASSES})
ENDIF(QT_WRAP_CPP)

ADD_LIBRARY(${SCRIBUS_BARCODEGENERATOR_PLUGIN} MODULE ${BARCODEGENERATOR_PLUGIN_SOURCES} ${BARCODEGENERATOR_PLUGIN_MOC_SOURCES} ${BARCODEGENERATOR_PLUGIN_UI_CPP})

INSTALL(TARGETS ${SCRIBUS_BARCODEGENERATOR_PLUGIN}
  LIBRARY
  DESTINATION ${PLUGINDIR}
  PERMISSIONS ${PLUGIN_PERMISSIONS}
)

INSTALL(FILES barcode.ps
  DESTINATION ${SHAREDIR}/plugins
)

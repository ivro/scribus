INCLUDE_DIRECTORIES(
${CMAKE_SOURCE_DIR}
${CMAKE_SOURCE_DIR}/scribus
)

barcodegenerator.ui
)

barcodegenerator.h
)

barcode.cpp
barcodegenerator.cpp
)

#IF(QT4_WRAP_UI)

#ENDIF(QT4_WRAP_UI)

#IF(QT4_WRAP_CPP)

  TARGET_LINK_LIBRARIES(${SCRIBUS_BARCODEGENERATOR_PLUGIN} "-undefined dynamic_lookup")

  DESTINATION ${PLUGINDIR}

)

  DESTINATION ${SHAREDIR}/plugins

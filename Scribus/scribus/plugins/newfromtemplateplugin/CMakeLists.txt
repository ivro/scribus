INCLUDE_DIRECTORIES(
${CMAKE_SOURCE_DIR}
${CMAKE_SOURCE_DIR}/scribus
)

SET(NEWFROMTEMPLATE_PLUGIN_MOC_CLASSES
nftdialog.h
nftemplate.h
)

SET(NEWFROMTEMPLATE_PLUGIN_SOURCES
nftdialog.cpp
nftemplate.cpp
nftrcreader.cpp
nftsettings.cpp
nfttemplate.cpp
)

SET(SCRIBUS_NEWFROMTEMPLATE_PLUGIN "newfromtemplateplugin")

IF(QT_WRAP_CPP)
  SCRIBUS_QT_WRAP_CPP(${NEWFROMTEMPLATE_PLUGIN_MOC_SOURCES} ${NEWFROMTEMPLATE_PLUGIN_MOC_CLASSES})
ENDIF(QT_WRAP_CPP)

ADD_LIBRARY(${SCRIBUS_NEWFROMTEMPLATE_PLUGIN} MODULE ${NEWFROMTEMPLATE_PLUGIN_SOURCES} ${NEWFROMTEMPLATE_PLUGIN_MOC_SOURCES})

INSTALL(TARGETS ${SCRIBUS_NEWFROMTEMPLATE_PLUGIN}
  LIBRARY
  DESTINATION ${PLUGINDIR}
  PERMISSIONS ${PLUGIN_PERMISSIONS}
)
